  <ParamField query='chainId' type='string' required default='bitcoin-testnet'>

  </ParamField>
  <ParamField body='jsonrpc' type='string' required default='2.0'>

  </ParamField>
  <ParamField body='method' type='string' required default='verifychain'>
    JSON-RPC method name.
  </ParamField>

<ParamField body='params' type='object' default={[0,0]} placeholder='[0,0]'>

  </ul>
