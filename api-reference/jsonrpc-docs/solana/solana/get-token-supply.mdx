### Parameters

<Expandable title='Default parameters'>
  <ParamField query='chainId' type='string' required default='solana'>
    Unique identifier for a blockchain network.
  </ParamField>
  <ParamField body='id' type='number' required default={1}>
    Request identifier used to match responses.
  </ParamField>
  <ParamField body='jsonrpc' type='string' required default='2.0'>
    JSON-RPC version.
  </ParamField>
  <ParamField body='method' type='string' required default='getTokenSupply'>
    JSON-RPC method name.
  </ParamField>
</Expandable>

<ParamField body='params' type='object' required placeholder='["3wyAj7Rt1TWVPZVteFJPLa26JmLvdb1CAKEFZm3NY75E"]'>
  Parameters for method getTokenSupply: [mintAddress, config]
  
  Positional parameters for <code>getTokenSupply</code>:
  <ul>
  <li><code>params[0]</code>: Pubkey of the token Mint to query (base-58 encoded string).</li>
  <li><code>params[1]</code>: Optional configuration object.</li>
  </ul>
</ParamField>

### Returns

<ResponseField name='result' type='object'>
  Token supply information for the given mint.
  <Expandable title='properties'>
    <ResponseField name='amount' type='string'>
      The raw total token supply without decimals (string representation of u64).
    </ResponseField>
    <ResponseField name='decimals' type='integer'>
      Number of base 10 digits to the right of the decimal place.
    </ResponseField>
    <ResponseField name='uiAmount' type='number'>
      Total token supply using mint-prescribed decimals (deprecated).
    </ResponseField>
    <ResponseField name='uiAmountString' type='string'>
      Total token supply as a string using mint-prescribed decimals.
    </ResponseField>
  </Expandable>
</ResponseField>
